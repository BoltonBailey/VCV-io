This pr is to add definitions pertaining to trees in /ToMathlib/Data/IndexedBinaryTree/Basic.lean

section Count

def Skeleton.leafCount : Skeleton → Nat
  | Skeleton.leaf => 1
  | Skeleton.internal left right =>
    left.leafCount + right.leafCount

end Count

section Depth

def Skeleton.depth : Skeleton → Nat
  | Skeleton.leaf => 0
  | Skeleton.internal left right =>
    Nat.max left.depth right.depth + 1

end Depth
